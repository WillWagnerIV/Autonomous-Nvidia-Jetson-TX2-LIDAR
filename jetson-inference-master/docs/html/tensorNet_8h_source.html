436  inline void PROFILER_BEGIN( profilerQuery query )
437  {
438  const uint32_t evt = query*2;
439  const uint32_t flag = (1 << query);
440 
441  CUDA(cudaEventRecord(mEventsGPU[evt], mStream));
442  timestamp(&mEventsCPU[evt]);
443 
444  mProfilerQueriesUsed |= flag;
445  mProfilerQueriesDone &= ~flag;
446  }
447 
451  inline void PROFILER_END( profilerQuery query )
452  {
453  const uint32_t evt = query*2+1;
454 
455  CUDA(cudaEventRecord(mEventsGPU[evt]));
456  timestamp(&mEventsCPU[evt]);
457  timespec cpuTime;
458  timeDiff(mEventsCPU[evt-1], mEventsCPU[evt], &cpuTime);
459  mProfilerTimes[query].x = timeFloat(cpuTime);
460 
461  if( mEnableProfiler && query == PROFILER_NETWORK )
462  {
463  printf(LOG_TRT "layer network time - %f ms\n", gProfiler.timingAccumulator);
464  gProfiler.timingAccumulator = 0.0f;
465  printf(LOG_TRT "note -- when processing a single image, run 'sudo jetson_clocks' before\n"
466  " to disable DVFS for more accurate profiling/timing measurements\n");
467  }
468  }

473  inline bool PROFILER_QUERY( profilerQuery query )
474  {
475  const uint32_t flag = (1 << query);
476 
477  if( query == PROFILER_TOTAL )
478  {
479  mProfilerTimes[PROFILER_TOTAL].x = 0.0f;
480  mProfilerTimes[PROFILER_TOTAL].y = 0.0f;
481 
482  for( uint32_t n=0; n < PROFILER_TOTAL; n++ )
483  {
484  if( PROFILER_QUERY((profilerQuery)n) )
485  {
486  mProfilerTimes[PROFILER_TOTAL].x += mProfilerTimes[n].x;
487  mProfilerTimes[PROFILER_TOTAL].y += mProfilerTimes[n].y;
488  }
489  }
490 
491  return true;
492  }
493  else if( mProfilerQueriesUsed & flag )
494  {
495  if( !(mProfilerQueriesDone & flag) )
496  {
497  const uint32_t evt = query*2;
498  float cuda_time = 0.0f;
499  CUDA(cudaEventElapsedTime(&cuda_time, mEventsGPU[evt], mEventsGPU[evt+1]));
500  mProfilerTimes[query].y = cuda_time;
501  mProfilerQueriesDone |= flag;
502  //mProfilerQueriesUsed &= ~flag;
503  }
504 
505  return true;
506  }
507 
508  return false;
509  }
